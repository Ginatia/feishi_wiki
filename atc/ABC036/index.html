    ABC036

    A - お茶

    一个箱子A个瓶子，要买B个瓶子需要买几个箱子

    向上取整即可

    void solve()
{
  int A,B;std::cin>>A>>B;
  std::cout<<(B+A-1)/A<<'\n';
}

    B - 回転

    有一个 N \times N 的方阵，将其顺时针旋转90度

    模拟即可

    void solve()
{
    int n;std::cin>>n;
    std::vector<std::string>a(n),b;
    b.reserve(n);
    for(auto&i:a)std::cin>>i;
    for(int j=0;j<n;j++)
    {
        std::string s;
        for(int i=0;i<n;i++)
        {
            s+=a[i][j];
        }
        std::reverse(s.begin(),s.end());
        b.push_back(s);
    }
    for(const auto&i:b){std::cout<<i<<"\n";}
}

    C - 座圧

    N个数，将数列 a_i 的值压缩为相对次序

    二分查找或map压缩

    void solve()
{
  int n;std::cin>>n;
  std::vector<int>a(n);
  for(auto&x:a){
    std::cin>>x;
  }
  auto v=a;
  std::sort(v.begin(),v.end());
  v.erase(std::unique(v.begin(),v.end()),v.end());

  for(auto x:a){
    std::cout<<std::lower_bound(v.begin(),v.end(),x)-v.begin()<<'\n';
  }
}

    D - 塗り絵

    有N个点的树，将点染成黑或者白，但不能出现一条边同为黑，取模 10^9+7

    树dp

    void solve()
{
  int N;std::cin>>N;
  std::vector<std::vector<int>>g(N);
  for(int i=1;i<N;i++){
    int a,b;
    std::cin>>a>>b;
    --a,--b;
    g[a].push_back(b);
    g[b].push_back(a);
  }
  std::vector<std::array<Z, 2>>dp(N,{Z(1),Z(1)});

  auto dfs=[&](auto self,int u,int f)->void{
    for(const auto&v:g[u]){
      if(v==f){
        continue;
      }
      self(self,v,u);
      dp[u][0]*=dp[v][0]+dp[v][1];
      dp[u][1]*=dp[v][0];
    }
  };

  dfs(dfs,0,-1);
  std::cout<<dp[0][0]+dp[0][1]<<'\n';
}
